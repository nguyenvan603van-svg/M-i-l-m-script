-- HÀM KHỞI CHẠY MAIN SCRIPT
local function StartMainScript()
    local text = Lang.Data[Lang.Current]

    local Window = Fluent:CreateWindow({
        Title = text.Title,
        SubTitle = text.Sub,
        TabWidth = 160,
        Size = UDim2.fromOffset(580, 460),
        Acrylic = true, 
        Theme = "Dark",
        MinimizeKey = Enum.KeyCode.LeftControl
    })

    -- Khởi tạo các Tab giống Maru Hub
    local Tabs = {
        Main = Window:AddTab({ Title = text.TabMain, Icon = "home" }),
        Combat = Window:AddTab({ Title = text.TabCombat, Icon = "swords" }),
        Stats = Window:AddTab({ Title = text.TabStats, Icon = "bar-chart-2" }),
        Teleport = Window:AddTab({ Title = text.TabTele, Icon = "map-pin" }),
        LangTab = Window:AddTab({ Title = text.TabLang, Icon = "globe" })
    }

    ---------------------------------------------------------
    -- [1] TAB MAIN / FARM
    ---------------------------------------------------------
    Tabs.Main:AddParagraph({ Title = "Maru Hub", Content = "Premium Blox Fruits Script" })

    -- Chọn vũ khí
    local WeaponDrop = Tabs.Main:AddDropdown("WeaponSelect", {
        Title = text.SelectWeapon,
        Values = {"Melee", "Sword", "Devil Fruit"},
        Default = "Melee",
        Callback = function(Value) _G.SelectWeapon = Value end
    })

    -- Auto Farm
    local FarmToggle = Tabs.Main:AddToggle("AutoFarm", {Title = text.AutoFarm, Default = false})
    FarmToggle:OnChanged(function(Value)
        _G.AutoFarm = Value
        while _G.AutoFarm do
            task.wait(0.5)
            print(Lang.Data[Lang.Current].FarmingStatus)
            -- Logic Farm Quái được thêm ở đây
        end
    end)

    -- Fast Attack & Bring Mob
    Tabs.Main:AddToggle("FastAttack", {Title = text.FastAttack, Default = false}):OnChanged(function(Value) _G.FastAttack = Value end)
    Tabs.Main:AddToggle("BringMob", {Title = text.BringMob, Default = false}):OnChanged(function(Value) _G.BringMob = Value end)

    ---------------------------------------------------------
    -- [2] TAB COMBAT
    ---------------------------------------------------------
    Tabs.Combat:AddToggle("KillAura", {Title = text.KillAura, Default = false}):OnChanged(function(Value) _G.KillAura = Value end)
    
    Tabs.Combat:AddDropdown("AimBotPlayer", {
        Title = "Aimbot Player (PVP)",
        Values = {"None"}, 
        Default = "None",
        Callback = function(Value) print("Target: " .. Value) end
    })

    ---------------------------------------------------------
    -- [3] TAB STATS (CỘNG ĐIỂM TIỀM NĂNG)
    ---------------------------------------------------------
    local StatPoints = {"Melee", "Defense", "Sword", "Gun", "Demon Fruit"}
    for _, stat in pairs(StatPoints) do
        Tabs.Stats:AddToggle("Stat_"..stat, {Title = text.AutoStat .. " " .. stat, Default = false}):OnChanged(function(Value)
            _G["Stat_"..stat] = Value
        end)
    end

    ---------------------------------------------------------
    -- [4] TAB TELEPORT (DỊCH CHUYỂN SEA)
    ---------------------------------------------------------
    for i = 1, 3 do
        Tabs.Teleport:AddButton({
            Title = text.TeleSea .. " " .. i,
            Callback = function()
                print("Teleporting to Sea " .. i)
            end
        })
    end

    ---------------------------------------------------------
    -- [5] TAB NGÔN NGỮ (CHỈNH TIẾNG ANH / TIẾNG VIỆT)
    ---------------------------------------------------------
    Tabs.LangTab:AddDropdown("LangSelect", {
        Title = text.ChangeLang,
        Values = {"Việt Nam", "English"},
        Default = Lang.Current,
        Callback = function(Value)
            if Lang.Current ~= Value then
                Lang.Current = Value
                -- Thông báo và tự động tải lại UI theo ngôn ngữ mới
                Fluent:Notify({
                    Title = "Language / Ngôn Ngữ",
                    Content = Value == "Việt Nam" and "Đã đổi sang Tiếng Việt!" or "Switched to English!",
                    Duration = 3
                })
                task.wait(0.5)
                Window:Destroy()   -- Xóa giao diện cũ
                StartMainScript()  -- Khởi chạy lại giao diện với ngôn ngữ đã chọn
            end
        end
    })

    -- Thông báo mở Menu thành công
    Fluent:Notify({
        Title = "Maru Hub",
        Content = text.NotifySuccess,
        Duration = 5
    })
end

-- CHẠY THẲNG VÀO SCRIPT KHÔNG QUA BƯỚC CHECK KEY
StartMainScript()
