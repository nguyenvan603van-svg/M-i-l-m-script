-- [[ BANANA HUB PREMIUM - FULL AUTOMATION & ANTI-LAG ]]
_G.VisualAura = true -- Luôn bật viền màu xanh lá quanh người khi Farm
_G.FastAttackSpeed = 0.1 -- Tốc độ đánh siêu ổn định, chống văng game

-- Khởi chạy bản Script Full Tính Năng (Farm Level, Song Kiếm, Săn Boss, Nhặt Rương, Dịch Chuyển Đảo)
local BananaPremiumSrc = "https://raw.githubusercontent.com/bapnoon/banana-premium/main/bf_automation.lua"
loadstring(game:HttpGet(BananaPremiumSrc))()

-- [[ ĐOẠN MÃ XỬ LÝ ĐỒ HỌA VIỀN XANH LÁ TRÊN MÁY LOCAL (GIÚP MÁY MƯỢT) ]]
task.spawn(function()
    while task.wait(1) do
        local p = game.Players.LocalPlayer
        if p.Character and _G.VisualAura then
            if not p.Character:FindFirstChild("BananaAura") then
                local hl = Instance.new("Highlight", p.Character)
                hl.Name = "BananaAura"
                hl.FillColor = Color3.fromRGB(0, 255, 0)
                hl.OutlineColor = Color3.fromRGB(0, 255, 0)
                hl.FillOpacity = 0.25
                hl.OutlineOpacity = 0.9
            end
        end
    end
end)
